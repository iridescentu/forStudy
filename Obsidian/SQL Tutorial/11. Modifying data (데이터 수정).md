- INSERT
```sql
INSERT INTO table1 (column1, column2,...)
VALUES
	(value1, value2,...);

# 여러 행 삽입
INSERT INTO table1
VALUES
	(value1, value2,...),
	(value1, value2,...),
	(value1, value2,...),
	...;

# 다른 테이블 행 복사
INSERT INTO table1 (column1, column2) 
SELECT
	column1,
	column2
FROM
	table2
WHERE
	condition1;

# * 사용 가능
INSERT INTO dependents_archive 
SELECT
	*
FROM
	dependents;
```

- UPDATE
```sql
UPDATE table_name
SET column1 = value1,
 column2 = value2
WHERE
	condition;

# 예시
UPDATE employees 
SET 
    last_name = 'Lopez'
WHERE
    employee_id = 192;
```

- DELETE
```sql
DELETE
FROM
	table_name
WHERE
	condition;
```
