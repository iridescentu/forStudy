- UNION
```sql
SELECT
	first_name,
	last_name
FROM
	employees
UNION
SELECT
	first_name,
	last_name
FROM
	dependents
ORDER BY
	last_name;
```

- EXCEPT
	* 마리아 DB 용
```sql
SELECT 
    employee_id
FROM
    employees 
EXCEPT 
SELECT 
    employee_id
FROM
    dependents
ORDER BY employee_id;




# 위와 동일
SELECT employee_id, dependent_id  
FROM employees 
LEFT JOIN dependents 
USING (employee_id)
WHERE dependent_id IS NULL 
ORDER BY employee_id;
```
